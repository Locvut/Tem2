#include<iostream>
#include <string>
using namespace std;
int main()
{
	string str;
	int key;
	cout << "enter the string: ";
	getline(cin, str);
	cout << "enter the key: ";
	cin >> key;
	for (int i = 0; i < str.length(); i++) 
	{
		if (str[i] == ' ') str.erase(i, 1);
	}	
	for (int i = 0; i < str.length(); i++)
	 {

		while (str[i] != '\0')
		{
		if (str[i] + key < 'z')
		str[i] = (str[i] + key);
		else
		str[i] = (str[i] - 26 + key);
		i++;
		}

	cout << "code string: " << str << '\n';
	return 0;
}
